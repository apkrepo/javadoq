0746 JAVACODE
0747 void node_descriptor_expression()
0748 {
0749   Token tok;
0750   int nesting = 1;
0751   while (true) {
0752     tok = getToken(1);
0753     if (tok.kind == 0) {
0754       throw new ParseException();
